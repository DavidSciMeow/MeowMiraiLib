45 private string PackMsg(int? syncid = null)
46 {
47 string s;
48 if (syncid != null)
49 {
50 syncId = syncid;
51 }
52 else
53 {
54 syncId = new Random().Next(int.MaxValue);
55 }
56 if (content is SMessage)
57 {
58 s = JsonConvert.SerializeObject(this, Formatting.None,
59 new JsonSerializerSettings { NullValueHandling = NullValueHandling.Ignore });
60 }
61 else
62 {
63 s = $"{{\"syncId\":{syncId},\"command\":\"{command}\"," +
64 $"{(subCommand != null ? $"\"subCommand\":\"{subCommand}\"," : "")}" +
65 $"\"content\":{content}}}";
66 }
67 return s;
68 }
76 public (bool isTimedOut, JObject? Return) Send(Client c, int? syncid = null, int TimeOut = 10)
77 {
78 this.session = c.session;
79 return c.SendAndWaitResponse(PackMsg(syncid), syncId, TimeOut).GetAwaiter().GetResult();
80 }
88 public async Task<(bool isTimedOut, JObject? Return)> SendAsync(Client c, int? syncid = null, int TimeOut = 10)
89 {
90 this.session = c.session;
91 return await c.SendAndWaitResponse(PackMsg(syncid), syncId, TimeOut);
92 }
